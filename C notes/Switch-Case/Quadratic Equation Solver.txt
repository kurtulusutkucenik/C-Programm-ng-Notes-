#include <stdio.h>
#include <math.h>

int main() {
    float root1, root2, discriminant, imaginary, a, b, c;

    printf("Enter the coefficients a, b, c of the quadratic equation ax^2 + bx + c: ");
    scanf("%f %f %f", &a, &b, &c);

    discriminant = (b * b) - (4 * a * c);
    // We compute discriminant first. We intentionally don't take sqrt yet
    // because sqrt of a negative number would be invalid.

    switch(discriminant > 0) {
        case 1: // Yes, discriminant is positive
            root1 = (-b - sqrt(discriminant)) / (2 * a);
            root2 = (-b + sqrt(discriminant)) / (2 * a);

            printf("There are two distinct real roots.\n");
            printf("Root 1 = %.2f \t Root 2 = %.2f\n", root1, root2);
            break;

        case 0: // Discriminant is not positive
            switch(discriminant < 0) {
                case 1: // Discriminant is negative → complex roots
                    root1 = -b / (2 * a); // Real part
                    imaginary = sqrt(-discriminant) / (2 * a); // Imaginary part

                    printf("There are two distinct complex roots.\n");
                    printf("Root 1 = %.2f + i%.2f \t Root 2 = %.2f - i%.2f\n", 
                           root1, imaginary, root1, imaginary);
                    break;

                case 0: // Discriminant = 0 → one real root
                    root1 = -b / (2 * a);

                    printf("There is a single real root.\n");
                    printf("Root = %.2f\n", root1);
                    break;

                default:
                    printf("Invalid value entered. Please check your input and try again.\n");
            }
            break;

        default:
            printf("Unexpected error occurred.\n");
    }

    return 0;
}

/* discriminant = b^2 - 4ac determines the nature of roots:

>0 → two distinct real roots

=0 → one real root (repeated)

<0 → two complex roots

sqrt() cannot handle negative numbers in real arithmetic, so we handle complex roots separately.

\t is a tab character, used to align output nicely.

Nested switch is used here to distinguish between zero and negative discriminant after checking for positive. */
