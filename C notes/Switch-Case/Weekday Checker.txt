#include <stdio.h>

int main() {
    int day;

    printf("Enter the day of the week (1-7): ");
    scanf("%d", &day);

    switch(day) {
        case 1:
        case 2:
        case 3:
        case 4:
        case 5:
            printf("Weekday\n");
            break;

        case 6:
        case 7:
            printf("Weekend\n");
            break;

        default:
            printf("Please enter a valid value (1-7)\n");
    }

    return 0;
}

/* Note:
 * In C, you **cannot** use the logical OR operator (||) in case labels like:
 *      case 1 || 2 || 3:
 * That would give a **compile-time error**.
 * The correct way is to list each value in a separate case as shown above.
 */
