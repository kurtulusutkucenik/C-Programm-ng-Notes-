#include <stdio.h>

int main() {

    int i;

    // Loop from 1 to 10
    for(i = 1; i <= 10; i++) {
        // Stop the loop when i equals 4
        if(i == 4) {
            break;  // Exit the loop immediately
        }
        printf("%d\n", i);
    }

    return 0;
}

/* This program demonstrates the break statement in a for loop:

Loop from i = 1 to i <= 10.

Check if i == 4.

If the condition is true, break exits the loop immediately.

Otherwise, print the value of i.

Output:

1
2
3  */
