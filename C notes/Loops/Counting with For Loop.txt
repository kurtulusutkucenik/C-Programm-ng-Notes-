#include <stdio.h>

int main() {

    int i, n;

    // Ask the user up to which number they want to count
    printf("Up to which number do you want to count?: ");
    scanf("%d", &n);

    // For loop: start at 1, continue until i <= n, increment i each iteration
    for(i = 1; i <= n; i++) {
        printf("%d\n", i);
    }

    return 0;
}

/* This program demonstrates counting from 1 to a user-specified number using a for loop:

Ask user for the upper limit: scanf("%d", &n);

for(i = 1; i <= n; i++)

i = 1 → starting value

i <= n → loop continues while condition is true

i++ → increment i by 1 each time

Print the value of i in each iteration  */
