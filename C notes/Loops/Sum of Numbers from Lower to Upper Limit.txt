#include <stdio.h>
int main(){
    int alt_sinir, ust_sinir, i, toplam=0;
    
    printf("Lütfen bir alt sınır giriniz: ");
    scanf("%d", &alt_sinir);
    
    printf("Lütfen bir üst sınır giriniz: ");
    scanf("%d", &ust_sinir);
    
    for(i = alt_sinir;i<=ust_sinir;i++){
        
        toplam += i;
    }
    printf("%d'den %d'ye olan sayıların toplamı = %d",alt_sinir,ust_sinir,toplam);
    return 0;
}
