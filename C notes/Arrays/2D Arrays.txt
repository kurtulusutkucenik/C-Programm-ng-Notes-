#include <stdio.h>

// 2D array (matrix): 2 rows, 3 columns
int matrix[2][3] = {
    {1, 2, 3},
    {4, 5, 6}
};

int main() {

    int i, j; // i -> row, j -> column

    for (i = 0; i < 2; i++) {
        for (j = 0; j < 3; j++) {
            printf("%d ", matrix[i][j]);
        }
        printf("\n");
    }

    return 0;
}
