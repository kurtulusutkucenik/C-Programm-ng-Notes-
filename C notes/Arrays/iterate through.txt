#include <stdio.h>

int main() {

    // Define an array with initial values
    int myArray[] = {25, 50, 75, 100};

    // Change the first element
    myArray[0] = 66;

    int i;

    // Traverse (iterate through) all elements of the array
    for(i = 0; i < 4; i++) {
        printf("%d ", myArray[i]);
    }

    return 0;
}

/* This program demonstrates how to traverse an array.

An array stores multiple values of the same type.

We use a for loop to access each element one by one.

The loop starts from index 0 because array indexing begins at 0.

The loop continues until i < 4, which is the number of elements.

Step-by-step:

Array created → {25, 50, 75, 100}

First element updated → {66, 50, 75, 100}

Loop prints all elements

OUTPUT = 66 50 75 100  */
