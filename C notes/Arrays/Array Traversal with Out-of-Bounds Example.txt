#include <stdio.h>

int main() {

    // Define an array of size 10, only first 4 elements initialized
    int myArray[10] = {25, 50, 75, 100};

    // Update the first element
    myArray[0] = 66;

    int i;

    // WARNING: Loop goes beyond array size!
    // Accessing elements outside the array bounds is undefined behavior
    for(i = 0; i < 13; i++) {
        printf("%d ", myArray[i]);
    }

    return 0;
}

/* The array has 10 elements: myArray[0] … myArray[9]

Only first 4 elements initialized: [25, 50, 75, 100, 0, 0, 0, 0, 0, 0]

Loop goes from i = 0 to i = 12 → indexes 10, 11, 12 are out-of-bounds

In C, accessing memory outside array bounds is undefined behavior

This may print garbage values or crash the program */
