// TRAVERSING ALL ELEMENTS IN AN ARRAY
#include <stdio.h>

int main() {

    // Define an array of size 10 (remaining elements default to 0)
    int myArray[10] = {25, 50, 75, 100};

    // Change the first element
    myArray[0] = 66;

    int i;

    // Iterate through all elements of the array
    for(i = 0; i < 10; i++) {
        printf("%d ", myArray[i]);
    }

    return 0;
}

/* This program demonstrates traversing (iterating through) all elements of an array. */
