#include <stdio.h>

// Program to remove spaces from a user-entered sentence and print it

int main(){
    
    char sentence[100];
    int i, j = 0;
    
    printf("Please enter a sentence: \n");
    scanf("%[^\n]", sentence);
    
    for(i = 0; sentence[i] != '\0'; i++){
        if(sentence[i] != ' '){
            sentence[j] = sentence[i];
            j++;
        }
    }
    sentence[j] = '\0';
    
    printf("Sentence without spaces: %s\n", sentence);
    
    return 0;
}
