#include <stdio.h>
#include <math.h>

// Program to print all even numbers up to a user-provided number

int main()
{
    int number, i;
    
    printf("Please enter a number: ");
    scanf("%d", &number);
    
    for(i = 0; i <= number; i++){
        if(i % 2 == 0){
            printf("%d ", i);
        }
    }
    return 0;
}
