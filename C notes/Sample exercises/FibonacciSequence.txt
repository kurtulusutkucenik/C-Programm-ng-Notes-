#include <stdio.h>

// Program to print Fibonacci sequence (C) using recursion

int fibonacci(int n);

int main(){
    int n1 = 0, n2 = 1, n3, count, i; // n1 and n2 initialized as the first two Fibonacci numbers
    
    printf("How many terms do you want? \n");
    scanf("%d", &count);
    
    printf("%d %d ", n1, n2);
    
    for(i = 2; i < count; i++){
        n3 = n1 + n2;
        printf("%d ", n3);
        n1 = n2; // previous second becomes first
        n2 = n3; // new sum becomes second
    }
    
    printf("\n");
    return 0;
}

int fibonacci(int n){
    if(n == 1)
        return 0;
    else if(n == 2)
        return 1;
    else
        return fibonacci(n-1) + fibonacci(n-2);
}
