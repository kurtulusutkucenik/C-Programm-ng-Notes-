#include <stdio.h>
#include <math.h>

// Program to check if a number is an Armstrong number
// For a number with m digits abc, it is Armstrong if abc = (a^m) + (b^m) + (c^m)

int isArmstrong(int number);

int main(){
    int number;
    
    printf("Enter a number: ");
    scanf("%d", &number);
    
    if(isArmstrong(number)){
        printf("It is an Armstrong number.\n");
    } else {
        printf("The entered number is not an Armstrong number.\n");
    }
    
    return 0;
}

int isArmstrong(int number)
{
    int original_number, num_digits, last_digit, sum;
    
    sum = 0;
    original_number = number;
    
    num_digits = (int) log10(number) + 1; // Calculate number of digits
    
    while(number > 0){
        last_digit = number % 10; // Get the last digit
        sum = sum + round(pow(last_digit, num_digits)); // Raise to power and round
        number = number / 10; // Remove last digit
    }
    
    return (original_number == sum);
}
