#include <stdio.h>
#include <stdlib.h>

int totalDuration(int durations[], int num_episodes) {
    int total = 0;
    for (int i = 0; i < num_episodes; i++){
        total += durations[i];
    }
    return total;
}

float averageDuration(int durations[], int num_episodes) {
    float sum = 0;
    for (int i = 0; i < num_episodes; i++) {
        sum += durations[i];
    }
    return sum / num_episodes;
}

void longEpisodes(int durations[], int num_episodes) {
    for (int i = 0; i < num_episodes; i++) {
        if (durations[i] >= 45) {
            printf("Long episode %d: %d minutes\n", i + 1, durations[i]); // i+1 because array index starts at 0
        }
    }
}

int main(void) {
    int num_episodes, minutes;

    printf("Enter the number of episodes: ");
    if (scanf("%d", &num_episodes) != 1) {
        printf("ERROR! Invalid input (not a number)\n");
        exit(EXIT_FAILURE);
    }

    int durations[num_episodes];

    for (int i = 0; i < num_episodes; i++) {
        printf("Enter the duration of episode %d (in minutes) and press Enter: ", i + 1);
        if (scanf("%d", &durations[i]) != 1) {
            printf("ERROR! You entered a non-numeric value.\n");
            exit(EXIT_FAILURE);
        }
    }

    printf("Total duration: %d minutes\n", totalDuration(durations, num_episodes));
    printf("Average duration: %.2f minutes\n", averageDuration(durations, num_episodes));

    longEpisodes(durations, num_episodes);

    return 0;
}
